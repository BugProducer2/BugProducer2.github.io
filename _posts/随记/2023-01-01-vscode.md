## 多个cpp文件

1. 配置launch.json文件。其中prelaunchtask是要接下来填写我们的tasks.json中配置的任务的。launch只是把编译好的文件调试而已。

```json
{
    // Use IntelliSense to learn about possible attributes.
    // Hover to view descriptions of existing attributes.
    // For more information, visit: https://go.microsoft.com/fwlink/?linkid=830387
    "version": "0.2.0",
    "configurations": [
        {
            "name": "(lldb) Launch",
            "type": "cppdbg",
            "request": "launch",
          	// program要有前面的fireDirname，否则会去根目录下找。
            "program": "${fileDirname}/${fileBasenameNoExtension}",
            "args": [],
            "stopAtEntry": false,
            "cwd": "${fileDirname}",
            "environment": [],
            "externalConsole": true,
            "MIMode": "lldb",
            "preLaunchTask": "C/C++: clang++ build active file"
        }

    ]
}
```



1. 配置tasks.json文件，需要指明需要编译哪些cpp文件。

```json
{
    "tasks": [
        {
            "type": "cppbuild",
            "label": "C/C++: clang++ build active file",
            "command": "/usr/bin/clang++", 
            "args": [
                "-fcolor-diagnostics",
                "-fansi-escape-codes",
                "-std=c++11",
              	// 生成和保留调试信息
                "-g",
                //"${file}",
              	// 需要编译的所有文件
                "keyrecovery4.cpp",
                "SmallAES.cpp",
                // "${fileDirname}/*.cpp",    // Compile all cpp files in the directory
                "-o",
                "${fileDirname}/${fileBasenameNoExtension}"
            ], 
            "options": {
                "cwd": "${fileDirname}"
            },
            "problemMatcher": [
                "$gcc"
            ],
            "group": {
                "kind": "build",
                "isDefault": true
            },
            "detail": "Task generated by Debugger."
        }
    ],
    "version": "2.0.0"
}
```
